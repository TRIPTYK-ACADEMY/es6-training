Manipulation des arrays avec find

Vous avez surement déjà utilisr une boucle for dans les cas ou un find sera juste parfait

        
                var users = [
                {name : 'Gilles'},
                {name : 'Joelle'},
                {name : 'Vincent'},
                ]
                var user;
                for (var i=0;i < users.length;i++){
                        if(users[i].name === 'Joelle'){
                                user= = users[i]  
                                break;  
                        } 
                }
